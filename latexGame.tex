% ── Helper: full-bleed coloured background ─────────────────
\newcommand{\pagebg}[1]{%
  \AddToShipoutPictureBG*{%
    \AtPageLowerLeft{%
      \color{#1}\rule{\paperwidth}{\paperheight}%
    }%
  }%
}

% ── Decorative TikZ border ─────────────────────────────────
\newcommand{\cornerdeco}[1]{%
\begin{tikzpicture}[remember picture, overlay]
  % thin outer frame
  \draw[color=#1, line width=0.6pt, opacity=0.55]
    ([xshift=18pt, yshift=-18pt]current page.north west) rectangle
    ([xshift=-18pt, yshift=18pt]current page.south east);
  % thicker inner frame
  \draw[color=#1, line width=1.2pt, opacity=0.25]
    ([xshift=24pt, yshift=-24pt]current page.north west) rectangle
    ([xshift=-24pt, yshift=24pt]current page.south east);
  % corner ornaments
  \foreach \anchor/\xmul/\ymul in {
      north west/1/-1, north east/-1/-1,
      south west/1/1,  south east/-1/1}{
    \begin{scope}[shift={(current page.\anchor)}]
      \draw[color=#1, line width=0.8pt, opacity=0.7]
        (\xmul*18pt, \ymul*18pt) -- (\xmul*38pt, \ymul*18pt)
        (\xmul*18pt, \ymul*18pt) -- (\xmul*18pt, \ymul*38pt);
      \node[color=#1, opacity=0.5, font=\tiny]
        at (\xmul*30pt, \ymul*30pt) {$\diamond$};
    \end{scope}
  }
\end{tikzpicture}%
}

% ── Scene title bar (top accent strip) ─────────────────────
\newcommand{\accentstrip}[1]{%
\begin{tikzpicture}[remember picture, overlay]
  \shade[left color=pagebg, middle color=#1, right color=pagebg,
         opacity=0.8]
    ([xshift=18pt, yshift=-18pt]current page.north west) rectangle
    ([xshift=-18pt, yshift=-21pt]current page.north east);
  \shade[left color=pagebg, middle color=#1, right color=pagebg,
         opacity=0.25]
    ([xshift=18pt, yshift=-22pt]current page.north west) rectangle
    ([xshift=-18pt, yshift=-26pt]current page.north east);
\end{tikzpicture}%
}

% ── Big centred icon ───────────────────────────────────────
\newcommand{\sceneicon}[1]{%
  \begin{center}
    \vspace{6pt}
    {\fontsize{56}{56}\selectfont #1}
    \vspace{-4pt}
  \end{center}%
}

% ── Thin rule with diamond ─────────────────────────────────
\newcommand{\goldrule}[1]{%
  \begin{center}
    \begin{tikzpicture}
      \draw[color=#1, line width=0.5pt, opacity=0.6]
        (0,0) -- (3.2cm, 0);
      \node[color=#1, opacity=0.8] at (3.7cm, 0) {$\diamond$};
      \draw[color=#1, line width=0.5pt, opacity=0.6]
        (4.2cm,0) -- (7.4cm, 0);
    \end{tikzpicture}
  \end{center}%
}

% ── Choice button macro ────────────────────────────────────
% #1 = accent colour   #2 = icon   #3 = label   #4 = link target
\newcommand{\choicebtn}[4]{%
  \vspace{6pt}
  \begin{mdframed}[
    backgroundcolor=#1!10!cardback,
    linecolor=#1,
    linewidth=0.7pt,
    innerleftmargin=14pt,
    innerrightmargin=14pt,
    innertopmargin=10pt,
    innerbottommargin=10pt,
    roundcorner=2pt,
    shadow=false
  ]
    \hyperlink{#4}{%
      \color{mistlight}%
      \textbf{#2}\quad
      {\addfontfeatures{}%
       \color{mistlight}\small #3}%
      \hfill{\color{#1}\small$\rightarrow$}%
    }
  \end{mdframed}%
  \vspace{2pt}%
}

  \end{center}%
}
